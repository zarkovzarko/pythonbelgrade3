    res = group[['name', 'births']].sort_values(by='births', ascending=False)[:3].reset_index()[['name', 'births']]
    res['ranking'] = res.index + 1
    return res

def plot_quartiles(year):
    global dfm
    dfmy = dfm[dfm.year == year]
    dfmy['quart'] = pd.qcut(dfmy.births, 4, labels=False)
    plt.figure()
    (dfmy.quart.value_counts().sort_index()/dfmy.shape[0]).plot.bar(rot=0, grid=True, title=year);
